# HydraSkript V1 Production Architecture Implementation Plan

## Core Technical Decisions

### Infrastructure Stack
```
Frontend: Next.js 15 App Router + Tailwind CSS + Zustand
Backend: Supabase (PostgreSQL + Realtime + Auth + Storage)
Database: PostgreSQL with pgvector extension
Queue: BullMQ on Redis
AI Orchestration: LangGraph
LLM Gateway: OpenRouter
Media Processing: FAL.ai, Gemini TTS, Cloudflare R2
Payments: Stripe Connect
Analytics: PostgreSQL Materialized Views
```

### Critical Design Principles
1. **Zero Data Loss**: All operations atomic with rollback mechanisms
2. **No Silent Failures**: Every job failure refunds credits immediately
3. **User Progress Always Preserved**: State persisted to DB with real-time sync
4. **Style-Centric Generation**: Every AI output passes through author's Style profile
5. **Continuity-First Approach**: Every generation checked against universe lore

```typescript
import { Worker, Job } from 'bullmq';
import redis from '@/lib/redis'; // Redis connection
import { refundCredits, deductCredits } from '@/lib/services/creditService';
import logger from '@/lib/utils/logger';

export abstract class BaseHydraWorker {
  protected worker: Worker;

  constructor(queueName: string) {
    this.worker = new Worker(queueName, async (job: Job) => {
      try {
        await this.processJob(job);

        // Deduct credits upon successful completion
        if (job.data.cost && !job.data.creditsDeducted) {
          await deductCredits(job.data.ownerId, job.data.cost, job.id);
          await job.update({ ...job.data, creditsDeducted: true });
        }

        logger.info(`Job ${job.id} completed successfully`);
      } catch (error) {
        logger.error(`Job ${job.id} failed:`, error);

        // Auto-refund credits on failure
        if (job.data.cost && !job.data.creditsRefunded) {
          await refundCredits(job.data.ownerId, job.data.cost, job.id);
          await job.update({ ...job.data, creditsRefunded: true });
        }

        throw error; // Re-throw to mark job as failed
      }
    }, { connection: redis });
  }

  protected abstract processJob(job: Job): Promise<void>;

  public async start(): Promise<void> {
    logger.info(`${this.constructor.name} started`);
  }

  public async close(): Promise<void> {
    await this.worker.close();
  }
}
```

```typescript
import { Queue, Job } from 'bullmq';
import redis from '@/lib/redis';
import { prisma } from '@/lib/database/prismaClient';
import logger from '@/lib/utils/logger';

class JobLifecycleManager {
  private static queues: Map<string, Queue> = new Map();

  static getQueue(queueName: string): Queue {
    if (!this.queues.has(queueName)) {
      const queue = new Queue(queueName, { connection: redis });
      this.queues.set(queueName, queue);
    }
    return this.queues.get(queueName)!;
  }

  static async createJob(
    ownerId: string,
    type: string,
    inputParams: any,
    cost: number
  ): Promise<Job> {
    // Validate credit balance before queuing
    const profile = await prisma.profile.findUnique({
      where: { id: ownerId }
    });

    if (!profile || profile.credit_balance < cost) {
      throw new Error('INSUFFICIENT_CREDITS');
    }

    // Create job record in database first
    const dbJob = await prisma.job.create({
      data: {
        owner_id: ownerId,
        type,
        status: 'queued',
        input_params: inputParams,
        cost,
        queued_at: new Date()
      }
    });

    // Add to BullMQ queue
    const queue = this.getQueue(type);
    const job = await queue.add(type, {
      jobId: dbJob.id,
      ownerId,
      type,
      inputParams,
      cost
    }, {
      jobId: dbJob.id
    });

    // Update position in queue
    await this.updateQueuePosition(dbJob.id);

    return job;
  }

  static async updateJobProgress(
    jobId: string,
    progressPercent: number,
    progressMessage: string
  ): Promise<void> {
    await prisma.job.update({
      where: { id: jobId },
      data: {
        progress_percent: progressPercent,
        progress_context_message: progressMessage,
        updated_at: new Date()
      }
    });

    // Emit real-time update via Supabase
    // This would be implemented with Supabase Realtime
  }

  static async completeJob(jobId: string, outputData: any): Promise<void> {
    const job = await prisma.job.findUnique({ where: { id: jobId } });

    if (!job) throw new Error('Job not found');

    await prisma.job.update({
      where: { id: jobId },
      data: {
        status: 'completed',
        output_data: outputData,
        finished_at: new Date(),
        updated_at: new Date()
      }
    });

    // Deduct credits
    if (job.cost && !job.credits_refunded) {
      await prisma.profile.update({
        where: { id: job.owner_id },
        data: {
          credit_balance: { decrement: job.cost }
        }
      });

      await prisma.creditTransaction.create({
        data: {
          profile_id: job.owner_id,
          amount: -job.cost,
          source_type: 'job_spend',
          source_id: jobId,
          balance_after: (await prisma.profile.findUnique({
            where: { id: job.owner_id }
          }))!.credit_balance
        }
      });
    }
  }

  static async failJob(jobId: string, errorMessage: string): Promise<void> {
    const job = await prisma.job.findUnique({ where: { id: jobId } });

    if (!job) throw new Error('Job not found');

    await prisma.job.update({
      where: { id: jobId },
      data: {
        status: 'failed',
        error_message: errorMessage,
        failed_at: new Date(),
        updated_at: new Date()
      }
    });

    // Auto-refund credits
    if (job.cost && !job.credits_refunded) {
      await prisma.profile.update({
        where: { id: job.owner_id },
        data: {
          credit_balance: { increment: job.cost }
        }
      });

      await prisma.creditTransaction.create({
        data: {
          profile_id: job.owner_id,
          amount: job.cost,
          source_type: 'job_refund',
          source_id: jobId,
          balance_after: (await prisma.profile.findUnique({
            where: { id: job.owner_id }
          }))!.credit_balance
        }
      });

      await prisma.job.update({
        where: { id: jobId },
        data: { credits_refunded: true }
      });
    }
  }

  static async cancelJob(jobId: string): Promise<void> {
    const job = await prisma.job.findUnique({ where: { id: jobId } });

    if (!job) throw new Error('Job not found');

    // Calculate pro-rata refund
    let refundAmount = job.cost || 0;
    if (job.progress_percent && job.progress_percent > 0) {
      refundAmount = Math.round(job.cost! * (100 - job.progress_percent) / 100);
    }

    await prisma.job.update({
      where: { id: jobId },
      data: {
        status: 'cancelled',
        cancel_requested: true,
        updated_at: new Date()
      }
    });

    // Refund credits
    if (refundAmount > 0 && !job.credits_refunded) {
      await prisma.profile.update({
        where: { id: job.owner_id },
        data: {
          credit_balance: { increment: refundAmount }
        }
      });

      await prisma.creditTransaction.create({
        data: {
          profile_id: job.owner_id,
          amount: refundAmount,
          source_type: 'job_refund',
          source_id: jobId,
          balance_after: (await prisma.profile.findUnique({
            where: { id: job.owner_id }
          }))!.credit_balance
        }
      });

      await prisma.job.update({
        where: { id: jobId },
        data: { credits_refunded: true }
      });
    }
  }

  static async updateQueuePosition(jobId: string): Promise<void> {
    // Simplified queue position calculation
    // In production, this would query Redis for actual position
    const job = await prisma.job.findUnique({ where: { id: jobId } });
    if (!job) return;

    const queueLength = await prisma.job.count({
      where: {
        type: job.type,
        status: 'queued',
        created_at: {
          lt: job.created_at
        }
      }
    });

    await prisma.job.update({
      where: { id: jobId },
      data: {
        position_in_queue: queueLength + 1
      }
    });
  }

  static async estimateETA(jobId: string): Promise<number> {
    // Simplified ETA estimation
    // In production, this would consider actual processing times
    const job = await prisma.job.findUnique({ where: { id: jobId } });
    if (!job || job.position_in_queue === null) return 0;

    // Assume 30 seconds per job in queue + processing time based on job type
    const baseTimePerJob = 30; // seconds
    const processingTime = this.getEstimatedProcessingTime(job.type);

    return (job.position_in_queue * baseTimePerJob) + processingTime;
  }

  private static getEstimatedProcessingTime(jobType: string): number {
    switch (jobType) {
      case 'style_training': return 120; // 2 minutes
      case 'chapter_generation': return 180; // 3 minutes
      case 'continuity_check': return 60; // 1 minute
      case 'audiobook_generation': return 300; // 5 minutes
      default: return 60;
    }
  }
}
```

```typescript
    const humanReadableIssues = await this.generateHumanReadableFixes(issues);

    };

  static async checkAndAutoFix(
    ownerId: string,
    universeId: string,
    chapterId: string,
    content: string,
    maxAttempts: number = 3,
    progressCallback?: (progress: number, message: string) => void
  ): Promise<{ success: boolean; content: string; issuesFixed: number }> {
    let currentContent = content;
    let totalIssuesFixed = 0;

    for (let attempt = 1; attempt <= maxAttempts; attempt++) {
      progressCallback?.(10 + (attempt * 20),
        `Continuity pass ${attempt} of ${maxAttempts}...`);

      const result = await this.checkContinuity(universeId, chapterId, currentContent);

      if (!result.hasIssues) {
        progressCallback?.(100, "Perfect continuity achieved!");
        return {
          success: true,
          content: currentContent,
          issuesFixed: totalIssuesFixed
        };
      }

      // Attempt to auto-fix issues
      const fixResult = await this.autoFixIssues(
        ownerId,
        universeId,
        chapterId,
        currentContent,
        result.issues
      );

      if (fixResult.fixed) {
        currentContent = fixResult.content;
        totalIssuesFixed += fixResult.issuesFixed;
        progressCallback?.(20 + (attempt * 20),
          `Auto-fixed ${fixResult.issuesFixed} issues in pass ${attempt}`);
      } else {
        // Could not auto-fix, break and report issues to user
        break;
      }
    }

    return {
      success: false,
      content: currentContent,
      issuesFixed: totalIssuesFixed
    };
  }
```

### Credit Service (`lib/services/creditService.ts`)
```typescript
import { prisma } from '@/lib/database/prismaClient';
import logger from '@/lib/utils/logger';

interface CreditTransactionInput {
  profileId: string;
  amount: number;
  sourceType: string;
  sourceId?: string;
}

export class CreditEngine {
  static async getCreditBalance(profileId: string): Promise<number> {
    const profile = await prisma.profile.findUnique({
      where: { id: profileId }
    });

    return profile?.credit_balance || 0;
  }

  static async getAvailableEarningActivities(profileId: string): Promise<any[]> {
    const activities = await prisma.creditEarningActivity.findMany({
      where: { is_active: true }
    });

    // Check progress toward each activity
    const activitiesWithProgress = await Promise.all(
      activities.map(async (activity) => {
        const progress = await this.checkActivityProgress(profileId, activity);
        return {
          ...activity,
          userProgress: progress.current,
          canEarn: progress.canEarn,
          nextAvailable: progress.nextAvailable
        };
      })
    );

    return activitiesWithProgress;
  }

  static async awardCredits(input: CreditTransactionInput): Promise<any> {
    // Start transaction
    const profile = await prisma.profile.findUnique({
      where: { id: input.profileId }
    });

    if (!profile) {
      throw new Error('Profile not found');
    }

    // Update profile balance
    const updatedProfile = await prisma.profile.update({
      where: { id: input.profileId },
      data: {
        credit_balance: { increment: input.amount },
        lifetime_credits_earned: { increment: Math.max(0, input.amount) }
      }
    });

    // Record transaction
    const transaction = await prisma.creditTransaction.create({
      data: {
        profile_id: input.profileId,
        amount: input.amount,
        source_type: input.sourceType,
        source_id: input.sourceId,
        balance_after: updatedProfile.credit_balance
      }
    });

    // Handle special cases
    if (input.sourceType === 'style_training_bonus') {
      await this.handleStyleTrainingBonus(input.profileId);
    }

    if (input.sourceType === 'first_generation') {
      await this.handleFirstGenerationBonus(input.profileId);
    }

    logger.info(`Awarded ${input.amount} credits to user ${input.profileId}`);

    return transaction;
  }

  static async deductCredits(
    profileId: string,
    amount: number,
    jobId?: string
  ): Promise<any> {
    const profile = await prisma.profile.findUnique({
      where: { id: profileId }
    });

    if (!profile || profile.credit_balance < amount) {
      throw new Error('INSUFFICIENT_CREDITS');
    }

    // Update profile balance
    const updatedProfile = await prisma.profile.update({
      where: { id: profileId },
      data: {
        credit_balance: { decrement: amount }
      }
    });

    // Record transaction
    const transaction = await prisma.creditTransaction.create({
      data: {
        profile_id: profileId,
        amount: -amount,
        source_type: 'job_spend',
        source_id: jobId,
        balance_after: updatedProfile.credit_balance
      }
    });

    return transaction;
  }

  static async refundCredits(
    profileId: string,
    amount: number,
    jobId?: string
  ): Promise<any> {
    // Update profile balance
    const updatedProfile = await prisma.profile.update({
      where: { id: profileId },
      data: {
        credit_balance: { increment: amount }
      }
    });

    // Record transaction
    const transaction = await prisma.creditTransaction.create({
      data: {
        profile_id: profileId,
        amount: amount,
        source_type: 'job_refund',
        source_id: jobId,
        balance_after: updatedProfile.credit_balance
      }
    });

    logger.info(`Refunded ${amount} credits to user ${profileId}`);

    return transaction;
  }

  static async getCreditContext(profileId: string): Promise<any> {
    const [balance, activities, transactions] = await Promise.all([
      this.getCreditBalance(profileId),
      this.getAvailableEarningActivities(profileId),
      prisma.creditTransaction.findMany({
        where: { profile_id: profileId },
        orderBy: { created_at: 'desc' },
        take: 20
      })
    ]);

    return {
      balance,
      available_earning_activities: activities,
      recent_transactions: transactions
    };
  }

  static async checkAndAwardCredits(profileId: string, activityType: string): Promise<boolean> {
    const activity = await prisma.creditEarningActivity.findUnique({
      where: { activity_type: activityType }
    });

    if (!activity || !activity.is_active) {
      return false;
    }

    // Check if user already earned this (for one-time activities)
    if (this.isOneTimeActivity(activity.activity_type)) {
      const existingTransaction = await prisma.creditTransaction.findFirst({
        where: {
          profile_id: profileId,
          source_type: 'daily_writing', // Map to correct source type
          created_at: {
            gte: new Date(Date.now() - (activity.cooldown_hours || 0) * 60 * 60 * 1000)
          }
        }
      });

      if (existingTransaction) {
        return false; // Already earned recently
      }
    }

    // Check progress toward threshold
    const progress = await this.checkActivityProgress(profileId, activity);

    if (progress.canEarn) {
      await this.awardCredits({
        profileId,
        amount: activity.credit_reward,
        sourceType: activity.activity_type.includes('daily') ? 'daily_writing' : activity.activity_type,
        sourceId: `activity_${activity.activity_type}`
      });

      return true;
    }

    return false;
  }

  private static async checkActivityProgress(profileId: string, activity: any): Promise<{
    current: any;
    canEarn: boolean;
    nextAvailable?: Date;
  }> {
    switch (activity.activity_type) {
      case 'daily_writing':
        const todayWords = await prisma.chapter.aggregate({
          where: {
            book: {
              universe: {
                owner_id: profileId
              }
            },
            created_at: {
              gte: new Date(new Date().setHours(0, 0, 0, 0))
            }
          },
          _sum: {
            word_count: true
          }
        });

        const currentWords = todayWords._sum.word_count || 0;
        const requirement = parseInt(activity.requirement_threshold || "500");

        return {
          current: `${currentWords} words`,
          canEarn: currentWords >= requirement,
          nextAvailable: currentWords >= requirement ? undefined : new Date()
        };

      case 'streak_3':
      case 'streak_5':
        const profile = await prisma.profile.findUnique({
          where: { id: profileId }
        });

        const streakDays = profile?.current_streak_days || 0;
        const requiredDays = activity.activity_type === 'streak_3' ? 3 : 5;

        return {
          current: `${streakDays} days`,
          canEarn: streakDays >= requiredDays,
          nextAvailable: streakDays >= requiredDays ? undefined : new Date()
        };

      case 'style_training':
        const styleCount = await prisma.style.count({
          where: {
            owner_id: profileId,
            training_status: 'ready'
          }
        });

        return {
          current: `${styleCount} trained styles`,
          canEarn: styleCount > 0,
          nextAvailable: styleCount > 0 ? undefined : new Date()
        };

      default:
        return {
          current: "0/1",
          canEarn: false
        };
    }
  }

  private static isOneTimeActivity(activityType: string): boolean {
    return [
      'style_training',
      'first_generation'
    ].includes(activityType);
  }

  private static async handleStyleTrainingBonus(profileId: string): Promise<void> {
    // Check if this is their first completed style training
    const completedStyles = await prisma.style.count({
      where: {
        owner_id: profileId,
        training_status: 'ready'
      }
    });

    if (completedStyles === 1) {
      // First style - award bonus
      await this.awardCredits({
        profileId,
        amount: 20,
        sourceType: 'style_training_bonus'
      });
    }
  }

  private static async handleFirstGenerationBonus(profileId: string): Promise<void> {
    // Check if this is their first AI generation
    const aiChapters = await prisma.chapter.count({
      where: {
        book: {
          universe: {
            owner_id: profileId
          }
        },
        generation_source: {
          in: ['ai_generated', 'ai_assisted']
        }
      }
    });

    if (aiChapters === 1) {
      // First generation - award bonus
      await this.awardCredits({
        profileId,
        amount: 10,
        sourceType: 'first_generation'
      });
    }
  }
}

// Export convenience functions
export async function getCreditBalance(profileId: string): Promise<number> {
  return CreditEngine.getCreditBalance(profileId);
}

export async function getCreditContext(profileId: string): Promise<any> {
  return CreditEngine.getCreditContext(profileId);
}

export async function awardCredits(input: CreditTransactionInput): Promise<any> {
  return CreditEngine.awardCredits(input);
}

export async function deductCredits(
  profileId: string,
  amount: number,
  jobId?: string
): Promise<any> {
  return CreditEngine.deductCredits(profileId, amount, jobId);
}

export async function refundCredits(
  profileId: string,
  amount: number,
  jobId?: string
): Promise<any> {
  return CreditEngine.refundCredits(profileId, amount, jobId);
}

export async function checkAndAwardCredits(profileId: string, activityType: string): Promise<boolean> {
  return CreditEngine.checkAndAwardCredits(profileId, activityType);
}
```

```typescript
  request: Request,
  { params }: { params: { id: string } }

    const session = await getServerSession();
    if (!session?.user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
```

```typescript
      return NextResponse.json({
        error: 'INSUFFICIENT_CREDITS',
        creditCost,
        currentBalance: balance

    }

    // Create job
    const job = await JobLifecycleManager.createJob(
      session.user.id,
      'chapter_generation',
      {
        chapterId,
        styleId,
        generationType,
        context,
        targetWordCount
      },
      creditCost
    );

    return NextResponse.json({
      jobId: job.id,
      queuePosition: 1, // Would be dynamic
      estimatedWaitTime: 180 // Would be calculated
    });
  } catch (error) {
    console.error('Chapter generation job creation error:', error);
```

```typescript
import { NextResponse } from 'next/server';
import { getServerSession } from '@supabase/auth-helpers-nextjs';
import { prisma } from '@/lib/database/prismaClient';

export async function GET(
  request: Request,
  { params }: { params: { id: string } }
) {
  try {
    const session = await getServerSession();
    if (!session?.user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }

    const universeId = params.id;
    const { searchParams } = new URL(request.url);
    const characterFilter = searchParams.get('character');

    // Verify universe ownership
    const universe = await prisma.universe.findUnique({
      where: { id: universeId }
    });

    if (!universe || universe.owner_id !== session.user.id) {
      return NextResponse.json({ error: 'Universe not found' }, { status: 404 });
    }

    // Get continuity events
    const whereClause: any = { universe_id: universeId };
    if (characterFilter) {
      whereClause.character_name = characterFilter;
    }

    const events = await prisma.continuityEvent.findMany({
      where: whereClause,
      orderBy: [
        { chapter_index: 'asc' },
        { created_at: 'asc' }
      ],
      include: {
        chapter: true
      }
    });

    // Group by character if no filter
    const groupedEvents: any = {};

    if (!characterFilter) {
      events.forEach(event => {
        const character = event.character_name || 'Unknown';
        if (!groupedEvents[character]) {
          groupedEvents[character] = [];
        }
        groupedEvents[character].push(event);
      });
    }

    // Get unresolved continuity issues for these events
    const eventIds = events.map(e => e.id);
    const issues = eventIds.length > 0
      ? await prisma.continuityIssue.findMany({
          where: {
            continuity_event_id: { in: eventIds },
            resolved: false
          }
        })
      : [];

    return NextResponse.json({
      events: characterFilter ? events : groupedEvents,
      unresolvedIssues: issues
    });
  } catch (error) {
    console.error('Get timeline error:', error);
    return NextResponse.json({ error: 'Failed to fetch timeline' }, { status: 500 });
  }
}
```

```typescript
    const session = await getServerSession();
```

```typescript
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }

    const chapterId = params.id;
    const {
      selectionStart,
      selectionEnd,
      assistType,
      selectionText,
      styleId
    } = await request.json();

    // Verify chapter ownership
    const chapter = await prisma.chapter.findUnique({
      where: { id: chapterId },
      include: { book: { include: { universe: true } } }
    });

    if (!chapter || chapter.book?.universe?.owner_id !== session.user.id) {

    let prompt = "";

        break;
      case 'deepen_description':
        prompt = `Enhance the descriptive details in this passage:\n\n${selectionText}\n\nRicher description:`;
        break;
      default:
        prompt = `Assist with this text:\n\n${selectionText}`;

    if (styleProfile && styleProfile.training_status === 'ready') {
      prompt = `
        Style Guide:
        - POV: ${styleProfile.pov_preference}
        - Tense: ${styleProfile.tense_preference}
        - Dialogue Ratio: ${Math.round(styleProfile.dialogue_ratio * 100)}%

        ${prompt}
      `;
    }

    // Generate assistance with Ollama
    const assistance = await generateTextWithOllama(prompt, {
      model: 'llama3:8b',
      temperature: 0.7,
      max_tokens: 300
    });

    return NextResponse.json({
      suggestedText: assistance,
```

### Base Workflow Class (`lib/langgraph/baseWorkflow.ts`)
```typescript
import { WorkflowState, NodeOutput } from './config';
import JobLifecycleManager from '@/lib/bullmq/jobLifecycleManager';

export abstract class BaseWorkflow {
  protected state: WorkflowState;

  constructor(initialState: WorkflowState) {
    this.state = initialState;
  }

  abstract execute(): Promise<any>;

  protected async updateProgress(progress: number, message: string): Promise<void> {
    this.state.progress = progress;
    this.state.progressMessage = message;

    await JobLifecycleManager.updateJobProgress(
      this.state.jobId,
      progress,
      message
    );
  }

  protected async handleError(error: any): Promise<void> {
    this.state.error = error.message;
    await JobLifecycleManager.failJob(this.state.jobId, error.message);
    throw error;
  }

  protected abstract validateInputs(): Promise<boolean>;
  protected abstract processNode(nodeName: string): Promise<NodeOutput>;
}
```

### Style Training Workflow (`lib/langgraph/workflows/styleTrainingWorkflow.ts`)
```typescript
import { BaseWorkflow, WorkflowState, NodeOutput } from '../baseWorkflow';
import { analyzeAuthorStyle } from '@/lib/services/styleTrainingService';
import { generateValidationSample } from '@/lib/services/styleTrainingService';
import { prisma } from '@/lib/database/prismaClient';

export class StyleTrainingWorkflow extends BaseWorkflow {
  async execute(): Promise<any> {
    try {
      // Validate inputs first
      const isValid = await this.validateInputs();
      if (!isValid) {
        throw new Error('Invalid inputs for style training');
      }

      await this.updateProgress(5, "Starting style analysis...");

      // Process each node in sequence
      let currentNode = 'validateInputs';
      let continueProcessing = true;

      while (continueProcessing && currentNode) {
        const nodeOutput = await this.processNode(currentNode);

        if (nodeOutput.nextState) {
          this.state = { ...this.state, ...nodeOutput.nextState };
        }

        if (nodeOutput.shouldContinue === false) {
          continueProcessing = false;
        }

        currentNode = nodeOutput.nextNode || '';
      }

      return this.state.result;
    } catch (error) {
      await this.handleError(error);
    }
  }

  protected async validateInputs(): Promise<boolean> {
    const { styleId, sampleTextPaths } = this.state.context;

    if (!styleId || !sampleTextPaths || sampleTextPaths.length < 3) {
      return false;
    }

    // Check if style exists
    const style = await prisma.style.findUnique({
      where: { id: styleId }
    });

    return !!style;
  }

  protected async processNode(nodeName: string): Promise<NodeOutput> {
    switch (nodeName) {
      case 'validateInputs':
        return await this.validateInputsNode();

      case 'analyzeSamples':
        return await this.analyzeSamplesNode();

      case 'extractPatterns':
        return await this.extractPatternsNode();

      case 'generateEmbedding':
        return await this.generateEmbeddingNode();

      case 'createValidationSample':
        return await this.createValidationSampleNode();

      case 'waitForApproval':
        return await this.waitForApprovalNode();

      default:
        throw new Error(`Unknown node: ${nodeName}`);
    }
  }

  private async validateInputsNode(): Promise<NodeOutput> {
    await this.updateProgress(10, "Validating training samples...");

    return {
      nextState: {
        step: 'analyzeSamples'
      },
      shouldContinue: true,
      nextNode: 'analyzeSamples'
    };
  }

  private async analyzeSamplesNode(): Promise<NodeOutput> {
    await this.updateProgress(20, "Analyzing your writing patterns...");

    const { styleId, sampleTextPaths } = this.state.context;

    const analysisResult = await analyzeAuthorStyle(
      this.state.ownerId,
      styleId,
      sampleTextPaths,
      async (progress, message) => {
        await this.updateProgress(20 + (progress * 0.3), message);
      }
    );

    return {
      nextState: {
```

```typescript
import { BaseWorkflow, WorkflowState, NodeOutput } from '../baseWorkflow';
import { generateChapterWithStyle } from '@/lib/services/chapterGenerationService';

import { prisma } from '@/lib/database/prismaClient';

export class ChapterGenerationWorkflow extends BaseWorkflow {
  async execute(): Promise<any> {
    try {
      const isValid = await this.validateInputs();
      if (!isValid) {
        throw new Error('Invalid inputs for chapter generation');
      }

      await this.updateProgress(5, "Preparing your story canvas...");

      let currentNode = 'prepareContext';
      let continueProcessing = true;

      while (continueProcessing && currentNode) {
        const nodeOutput = await this.processNode(currentNode);

        if (nodeOutput.nextState) {
          this.state = { ...this.state, ...nodeOutput.nextState };
        }

        if (nodeOutput.shouldContinue === false) {
          continueProcessing = false;
        }

        currentNode = nodeOutput.nextNode || '';
      }

      return this.state.result;
    } catch (error) {
      await this.handleError(error);
    }
  }

  protected async validateInputs(): Promise<boolean> {
    const { bookId, styleId, context, targetWordCount } = this.state.context;

    if (!bookId || !styleId || !context || !targetWordCount) {
      return false;
    }

    // Validate book and style exist
    const book = await prisma.book.findUnique({
      where: { id: bookId }
    });

    const style = await prisma.style.findUnique({
      where: { id: styleId }
    });

    return !!(book && style && style.training_status === 'ready');
  }

  protected async processNode(nodeName: string): Promise<NodeOutput> {
    switch (nodeName) {
      case 'prepareContext':
        return await this.prepareContextNode();

      case 'retrieveStyle':
        return await this.retrieveStyleNode();

      case 'fetchUniverseContext':
        return await this.fetchUniverseContextNode();

      case 'generateContent':
        return await this.generateContentNode();

      case 'checkContinuity':
        return await this.checkContinuityNode();

      case 'saveChapter':
        return await this.saveChapterNode();

      default:
        throw new Error(`Unknown node: ${nodeName}`);
    }
  }

  private async prepareContextNode(): Promise<NodeOutput> {
    await this.updateProgress(10, "Setting up your creative environment...");

    return {
      nextState: {
        step: 'retrieveStyle'
      },
      shouldContinue: true,
      nextNode: 'retrieveStyle'
    };
  }

  private async retrieveStyleNode(): Promise<NodeOutput> {
    await this.updateProgress(15, "Loading your unique writing style...");

    const { styleId } = this.state.context;
    const style = await prisma.style.findUnique({
      where: { id: styleId }
    });

    if (!style) {
      throw new Error('Style not found');
    }

    return {
      nextState: {
        step: 'fetchUniverseContext',
        context: {
          ...this.state.context,
          style
        }
      },
      shouldContinue: true,
      nextNode: 'fetchUniverseContext'
    };

  private async fetchUniverseContextNode(): Promise<NodeOutput> {

      nextState: {
        step: 'generateContent'
      },
      shouldContinue: true,
      nextNode: 'generateContent'
    };
  }

  private async generateContentNode(): Promise<NodeOutput> {

    const { bookId, styleId, context, targetWordCount } = this.state.context;

    const result = await generateChapterWithStyle(
      this.state.ownerId,
      styleId,
      {
        bookId,
        outline: context.outline,
        direction: context.direction,
        tone: context.tone,
        targetWordCount,
        previousChapterId: context.previousChapterId
      },

        step: 'checkContinuity',

        await this.updateProgress(70 + (progress * 0.2), message);
      }
    );

    return {
      nextState: {
        step: 'saveChapter',
        context: {
          ...this.state.context,
          continuityResult
        }
      },
      shouldContinue: true,
      nextNode: 'saveChapter'
    };
  }

  private async saveChapterNode(): Promise<NodeOutput> {
```
